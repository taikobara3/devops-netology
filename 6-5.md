* текст Dockerfile манифеста:
```
version: "3.3"

services:

  netology_test:
    image: elasticsearch:7.17.7
    environment:
      - http.host=0.0.0.0
      - transport.host=0.0.0.0
      - path.data=/var/lib/es
      - discovery.type=single-node
      - node.name=netology_test
    ports:
      - 9200:9200
      - 9300:9300
    volumes:
      - esdata:/var/lib/es:rw
    container_name: netology_test

volumes:
  esdata:
     driver: local
```
